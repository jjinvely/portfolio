02. 수학 메서드 : Math.random() : 0과 1사이의 난수 값을 반환 : 반환(숫자)

번호 기본값 메서드 리턴값 클릭
1 0 ~ 1 Math.random() 0.2936526420852983
2 0 ~ 9 Math.random() 2
2 10 ~ 20 Math.random() 17 
document.querySelector(".sample2_N1").innerHTML = 1;
document.querySelector(".sample2_Q1").innerHTML = "0 ~ 1";
document.querySelector(".sample2_M1").innerHTML = "Math.random()";
document.querySelector(".sample2_P1").innerHTML = Math.random();
document.querySelector(".sample2_P1").style.color = "black";
document.querySelector(".sample2_B1").addEventListener("click", () => {
    document.querySelector(".sample2_P1").innerHTML = Math.random();
})

document.querySelector(".sample2_N2").innerHTML = 2;
document.querySelector(".sample2_Q2").innerHTML = "0 ~ 9";
document.querySelector(".sample2_M2").innerHTML = "Math.random()";
document.querySelector(".sample2_P2").innerHTML = Math.floor(Math.random() * 10);
document.querySelector(".sample2_P2").style.color = "black";
document.querySelector(".sample2_B2").addEventListener("click", () => {
    document.querySelector(".sample2_P2").innerHTML = Math.floor(Math.random() * 10);
})

document.querySelector(".sample2_N3").innerHTML = 2;
document.querySelector(".sample2_Q3").innerHTML = "10 ~ 20";
document.querySelector(".sample2_M3").innerHTML = "Math.random()";
document.querySelector(".sample2_P3").innerHTML = 10 + Math.floor(Math.random() * 10);
document.querySelector(".sample2_P3").style.color = "black";
document.querySelector(".sample2_B3").addEventListener("click", () => {
    document.querySelector(".sample2_P3").innerHTML = 10 + Math.floor(Math.random() * 10);
})
